    return 1
  fi
  
  local branch=$1
  local dir="../${PWD##*/}-$branch"
  
  # ë¸Œëœì¹˜ ì¡´ì¬ ì—¬ë¶€ í™•ì¸

function wr() {
  if [ -z "$1" ]; then
    echo "ì‚¬ìš©ë²•: wr <ì›Œí¬íŠ¸ë¦¬ëª…>"
    return 1
  fi
  
  local name=$1
  local main_dir=$(git worktree list | head -1 | awk '{print $1}')

## completions
_git_branch_completion() {
    local -a branches
    branches=(${(f)"$(git branch --format='%(refname:short)')"})
    compadd -M 'm:{[:lower:][:upper:]}={[:upper:][:lower:]} l:|=* r:|=*' "$@" $branches
}
